  <h2 className="text-3xl font-semibold text-gray-900 dark:text-white mb-3">Quick start</h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">Get up and running with SEPA XML generation in minutes</p>
  
  <Steps>
    <Step title="Install the package">
      Install sepa-js-xml using your preferred package manager:
      
      <CodeGroup>
      ```bash npm
      npm install sepa-js-xml
      ```
      ```bash yarn
      yarn add sepa-js-xml
      ```
      ```bash pnpm
      pnpm add sepa-js-xml
      ```
      ```bash bun
      bun add sepa-js-xml
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Import and configure">
      Import the library and prepare your payment data:
      
      ```typescript
      import { createSepaXML } from "sepa-js-xml";
      ```
    </Step>
    
    <Step title="Generate your first SEPA XML">
      Create a SEPA XML file with your payment information:
      
      ```typescript
      const xml = createSepaXML({
        id: "MSG-001",
        creationDate: new Date(),
        initiatorName: "Your Company Name",
        positions: [
          {
            id: "PMT-001",
            name: "Debtor Account",
            iban: "DE89370400440532013000",
            bic: "COBADEFFXXX",
            requestedExecutionDate: new Date(),
            payments: [
              {
                id: "TXN-001",
                name: "Creditor Name",
                iban: "DE89370400440532013001",
                bic: "COBADEFFXXX",
                amount: 100.50,
                remittanceInformation: "Invoice #12345"
              }
            ]
          }
        ]
      });
      
      console.log(xml);
      ```
      
      <Note>The library validates IBANs and BICs by default. You can disable validation by passing options.</Note>
    </Step>
  </Steps>
</div>
